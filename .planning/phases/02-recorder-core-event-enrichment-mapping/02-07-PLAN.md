    """Reusable mock from Plan 05; included here for self-containment."""

    def __init__(self, events: list[AgentEvent]) -> None:
        self._events = events

    def execute(self, *args, **kwargs) -> AsyncIterator[AgentEvent]:
        async def _gen() -> AsyncIterator[AgentEvent]:
            for e in self._events:
                yield e
        return _gen()

        return None
